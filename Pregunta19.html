import pandas as pd
import numpy as np
from google.colab import files
import statsmodels.api as sm
from statsmodels.tsa.stattools import adfuller, kpss
from statsmodels.tsa.vector_ar.vecm import coint_johansen
import matplotlib.pyplot as plt

# Paso 1: Cargar los datos
def load_data():
    print("Por favor, sube el archivo Excel que contiene los datos de TB3M y TB6M.")
    upload = files.upload()
    filename = list(upload.keys())[0]
    try:
        xls = pd.ExcelFile(filename)
        if len(xls.sheet_names) == 0:
            raise ValueError("El archivo Excel no contiene hojas.")
        df = pd.read_excel(xls, sheet_name=0)  # Leer la primera hoja
    except ValueError as e:
        print(f"Error: {e}")
        return None
    except Exception as e:
        print(f"Error al leer el archivo: {e}")
        return None
    return df

# Paso 2: Verificar las columnas necesarias
def check_columns(df):
    required_columns = ['Date', 'TB3M', 'TB6M']
    for column in required_columns:
        if column not in df.columns:
            print(f"Error: La columna '{column}' no se encuentra en el DataFrame.")
            return False
    return True

# Paso 3: Graficar las series temporales
def plot_time_series(df):
    plt.figure(figsize=(12, 6))
    plt.plot(df['Date'], df['TB3M'], label='TB3M')
    plt.plot(df['Date'], df['TB6M'], label='TB6M')
    plt.title('Tasas de Letras del Tesoro a Tres y Seis Meses')
    plt.xlabel('Fecha')
    plt.ylabel('Tasa')
    plt.legend()
    plt.show()

# Paso 4: Pruebas de raz unitaria
def unit_root_tests(series, name):
    print(f"\nPruebas de raz unitaria para {name}:")
    adf_result = adfuller(series)
    print("Prueba ADF:")
    print(f'Estadstico ADF: {adf_result[0]:.4f}, p-valor: {adf_result[1]:.4f}')
    print("Interpretacin:", "Estacionario" if adf_result[1] < 0.05 else "No estacionario")

    kpss_result = kpss(series)
    print("\nPrueba KPSS:")
    print(f'Estadstico KPSS: {kpss_result[0]:.4f}, p-valor: {kpss_result[1]:.4f}')
    print("Interpretacin:", "No estacionario" if kpss_result[1] < 0.05 else "Estacionario")

# Paso 5: Prueba de cointegracin
def cointegration_test(df):
    result = coint_johansen(df[['TB3M', 'TB6M']], det_order=0, k_ar_diff=1)
    print("\nPrueba de Cointegracin de Johansen:")
    print(f'Estadsticos de traza: {result.lr1}')
    print(f'Valores crticos (95%): {result.cvt[:, 1]}')
    if result.lr1[0] > result.cvt[0, 1]:
        print("Conclusin: Las series estn cointegradas.")
        return True
    else:
        print("Conclusin: Las series no estn cointegradas.")
        return False

# Paso 6: Interpretacin econmica de la cointegracin
def economic_interpretation_cointegration(is_cointegrated):
    if is_cointegrated:
        print("\nInterpretacin Econmica:")
        print("La cointegracin entre TB3M y TB6M sugiere que hay una relacin de equilibrio a largo plazo entre las dos tasas de letras del Tesoro.")
        print("Esto significa que si una tasa se desva de la otra, eventualmente volvern a moverse juntas.")
    else:
        print("\nInterpretacin Econmica:")
        print("Si las series no estn cointegradas, implica que no hay una relacin de equilibrio a largo plazo entre las tasas de TB3M y TB6M.")
        print("Esto significa que las dos tasas pueden moverse independientemente una de la otra a largo plazo.")

# Paso 7: Anlisis VAR
def var_analysis(df, is_cointegrated):
    if is_cointegrated:
        print("\nAnlisis VAR en niveles:")
        print("Dado que las series estn cointegradas, se puede realizar el anlisis VAR en niveles.")
    else:
        print("\nAnlisis VAR en primeras diferencias:")
        print("Dado que las series no estn cointegradas, se recomienda realizar el anlisis VAR en primeras diferencias para evitar resultados espurios.")

# Funcin principal
def main():
    # Cargar datos
    df = load_data()
    if df is None:
        return

    # Verificar columnas
    if not check_columns(df):
        return

    # Graficar series temporales
    plot_time_series(df)

    # Pruebas de raz unitaria
    unit_root_tests(df['TB3M'], 'TB3M')
    unit_root_tests(df['TB6M'], 'TB6M')

    # Prueba de cointegracin
    is_cointegrated = cointegration_test(df)

    # Interpretacin econmica
    economic_interpretation_cointegration(is_cointegrated)

    # Anlisis VAR
    var_analysis(df, is_cointegrated)

# Ejecutar la funcin principal
if __name__ == "__main__":
    main()
